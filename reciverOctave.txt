function reciverInHex()
  archivo_operacion = 'operacion.txt';

  while true
    if exist(archivo_operacion, 'file')
      operacion = fileread(archivo_operacion);
      printf("Operaci√≥n recibida: %s\n", operacion);
      resultado = evalc(operacion);
      printf("Resultado en palabras: %s\n", resultado);

      delete(archivo_operacion);
    else
      pause(1);
    end
  end
endfunction
